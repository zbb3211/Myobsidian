- 二分找每个数对应需求的数
- `memcpy`用于复制数组
```c++
memcpy(w, q, sizeof w);  // 将q复制到w

for (int i = 0; i < n; i ++ )
{
	int l = 0, r = n - 1;
	while (l < r)
	{
		int mid = l + r + 1 >> 1;
		if (w[mid] <= q[i] * 2) l = mid;
		else r = mid - 1;
	}

	if (w[r] == q[i]) r -- ;
	if (r < 0) printf("-1 ");
	else printf("%d ", w[r]);
}

```
